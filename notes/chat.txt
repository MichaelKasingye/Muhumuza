const convertHistory = (oldHistory: { [key: string]: string }[] | any) => {
    return oldHistory.map((entry:any) => {
      const role = Object.keys(entry)[0] === 'Human' ? 'user' : 'assistant';
      return { role, content: entry[role === 'user' ? 'Human' : 'AI'] };
    });
  };

export default function Home() {
  const [message, setMessage] = useState<string | any>("");
  const [preChats, setPreChats] = useState<string | any>("");
  const [history, setHistory] = useState<Message[] | any>([
    {
      role: "assistant",
      content:
        "Hello! Ask me legal questions about immigration to the Netherlands.",
    },
  ]);
  const { user,signInWithGoogle, signOut } = useAuth();

  const lastMessageRef = useRef<HTMLDivElement | null>(null);
  const [loading, setLoading] = useState<boolean>(false);


  const [showModal, setShowModal] = useState(false);
  const chatHistory: ChatMessage[] = [
    { sender: 'Bot', text: 'Hello! How can I assist you today?' },
    { sender: 'User', text: 'I need help with my account.' },
    { sender: 'Bot', text: 'Sure, let me check your account details.' },
    // Add more chat messages here
  ];

  // console.log("user- chat",user);
  // console.log("signInWithGoogle- chat",signInWithGoogle);
  // 

  const handleClick = () => {
    if (message == "") return;
    setHistory((oldHistory:any) => [
      ...oldHistory,
      { role: "user", content: message },
    ]);
    setMessage("");
    setLoading(true);
    fetch("/api/chat", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ query: message, history: history, user }),
    })
      .then(async (res) => {
        const resonseData = await res.json();
        // console.log('resonseData',resonseData.historyData);
       const convertedData = convertHistory(resonseData.historyData)
    //    console.log('convertedData',convertedData);
       console.log('resonseData',resonseData);

        setHistory((oldHistory:any) => [...oldHistory, resonseData]);
        setPreChats((oldHistory:any) => [...oldHistory, convertedData]);
        setLoading(false);
      })
      .catch((err) => {
        alert(err);
      });
  };

  const formatPageName = (url: string) => {
    // Split the URL by "/" and get the last segment
    const pageName = url.split("/").pop();

    // Split by "-" and then join with space
    if (pageName) {
      const formattedName = pageName.split("-").join(" ");

      // Capitalize only the first letter of the entire string
      return formattedName.charAt(0).toUpperCase() + formattedName.slice(1);
    }
  };
// console.log('preChats',preChats);
useEffect(() => {

  // const unsubscribe = auth.onAuthStateChanged((user: any) => {
    if (user) {
      console.log('user-> chat',user);
      
  // router.push('/');
} else {
      return
    }
    // setLoading(false);
  // });

  // return () => unsubscribe();
}, [user]); 
  //scroll to bottom of chat
  useEffect(() => {
    if (lastMessageRef.current) {
      lastMessageRef.current.scrollIntoView({ behavior: "smooth" });
    }
    // console.log('message',message.content);
    
    // setHistory(message.historyData)
  }, [history]);

  return (
    <main className="h-screen bg-white p-6 flex flex-col">
      <div className="flex flex-col gap-8 w-full items-center flex-grow max-h-full">
        <h1 className=" text-4xl text-white font-extralight bg-clip-text">
          Muhumuza AI
        </h1>
      
        <form
          className="rounded-2xl border-purple-700 border-opacity-5  border lg:w-3/4 flex-grow flex flex-col bg-[url('/images/bg.png')] bg-cover max-h-full overflow-clip"
          onSubmit={(e) => {
            e.preventDefault();
            handleClick();
          }}
        >



<div className=" flex items-center justify-center bg-gray-100">
      <button
        className="bg-blue-500 text-white px-4 py-2 rounded hover:bg-blue-700"
        onClick={() => setShowModal(true)}
      >
        Show Chat History
      </button>

      <Modal showModal={showModal} setShowModal={setShowModal}>
        {/* <ChatHistory chatHistory={chatHistory} /> */}
        <ChatHistory chatHistory={[...preChats]} />
      </Modal>
    </div>



          <div className="overflow-y-scroll flex flex-col gap-5 p-10 h-full">
{/* History */}
   
{[...preChats].map((message: Message | any, idx:any) => {
    // console.log('jsx log',message);
    // console.log('jsx log',message.map((item:any) => item.role));
    message.map((messageContent:any, index:any) => {
    // console.log('jsx messageContent',messageContent.role);
        
        // const isLastMessage = index === history.length - 1;
        // console.log('isLastMessage',isLastMessage);
        <div>
            <h1>{messageContent.role}</h1>
        </div>
        switch (messageContent.role) {
          case "assistant":
            return (
              <div

                key={index}
                className="flex gap-2"
              >
                <img

              <div
                className="w-auto max-w-xl break-words bg-white rounded-b-xl rounded-tl-xl text-black p-6 self-end shadow-[0_10px_40px_0px_rgba(0,0,0,0.15)]"

                <p className="text-sm font-medium text-violet-500 mb-2">
                  You
                </p>
                {messageContent.content}
              </div>

        }

    })
            }
            )
            }


{/* Chat */}
            {history.map((message: Message, idx:any) => {
              const isLastMessage = idx === history.length - 1;
              switch (message.role) {
                case "assistant":

                    <div
                      ref={isLastMessage ? lastMessageRef : null}
                      key={idx}

                    >
                      <img

                        className="h-12 w-12 rounded-full"

                      <div className="w-auto max-w-xl break-words bg-white rounded-b-xl rounded-tr-xl text-black p-6 shadow-[0_ 10px_40px_0px_rgba(0,0,0,0.15)]">
